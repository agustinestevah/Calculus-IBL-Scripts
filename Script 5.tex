\begin{ax} \textbf{Axiom 4}\\
\label{Axiom 4}
A continuum is connected.
\end{ax}
\subsection{Theorem 5.2}
\begin{thm}
The only subsets of a continuum $C$ that are both open and closed are $\emptyset$ and~$C$.
\end{thm}
\begin{proof}:\\
    \begin{enumerate} {}
    \item $C$ is closed by Theorem 4.2. $C$ is open by Theorem 4.8.
    \item $\emptyset$ is closed by Theorem 4.2. $\emptyset$ is open by Theorem 4.8.
    \item Assume, for the sake of contradiction, that there exists some $A\subset C$ such that $A$ is both open and closed. Therefore, because $A$ is closed, then $C\setminus A$ is open. Moreover $C = A \cup C\setminus A$. $A$ is nonempty because it is not equal to the emptyset and therefore contains some element $x$. $C\setminus A$ is nonempty because $A\neq C$ and therefore there exists some $q\in C$ such that $q\notin A$. Therefore, because $A, C\setminus A$ are both open, disjoint, and nonempty, and their union is equal to the continuum, then $C$ is disconnected by Definition 4.22. However, that is a contradiction to Axiom \ref{Axiom 4}. Therefore, there does not exist some $A \neq C,\emptyset$ such that $A$ is both open and closed.
    \end{enumerate}
\end{proof}
